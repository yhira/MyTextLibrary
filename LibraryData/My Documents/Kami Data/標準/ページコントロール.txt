procedure TForm1.FormCreate(Sender: TObject);
begin
  ListBox2.Items := Screen.Fonts;
  UpDown1.Position := Label1.Font.Size;
end;

procedure TForm1.ListBox2Click(Sender: TObject);
begin
  Label1.Font.Name := ListBox2.Items[ListBox2.itemindex];
end;

procedure TForm1.UpDown1Changing(Sender: TObject;
  var AllowChange: Boolean);
begin
  Label1.Font.Size := UpDown1.Position;
end;

procedure TForm1.Edit1KeyPress(Sender: TObject; var Key: Char);
begin
  if Key = #13 then
  begin
    Label1.Font.Size := StrToInt(Edit1.Text);
    Key := #0;
  end;

end;

procedure TForm1.ListBox1Click(Sender: TObject);
begin
  Label1.Font.Color := StringToColor(ListBox1.Items[ListBox1.itemindex]);
end;
